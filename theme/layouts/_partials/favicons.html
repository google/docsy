{{- /* Renders favicon tags and publishes the files at the site root, where browsers and other clients probe for them by convention (e.g. /favicon.ico, /apple-touch-icon.png). The theme ships no favicon files; sites provide their own under `assets/`, at the root or in a `favicons/` directory, and may override behavior via `params.td.favicons` (`path`, `sizes`, `no_apple_precomposed`). A single square source image is resized to each configured size; optional `.svg` and `.ico` variants are published as-is. See https://www.docsy.dev/docs/content/iconsimages/#add-your-favicons */ -}} {{- $cfg := index (.Site.Params.td | default dict) "favicons" | default dict -}} {{- $sizes := or (index $cfg "sizes") (slice 16 32 48) -}} {{- $noApplePrecomposed := index $cfg "no_apple_precomposed" -}} {{- $base := or (index $cfg "path") "favicon.*" -}} {{- with path.Ext $base }}{{ $base = strings.TrimSuffix . $base }}{{ end -}} {{- $name := path.Base $base -}} {{- $svg := or (resources.Get (printf "%s.svg" $base)) (resources.Get (printf "favicons/%s.svg" $name)) -}} {{- $ico := or (resources.Get (printf "%s.ico" $base)) (resources.Get (printf "favicons/%s.ico" $name)) -}} {{- $source := or (resources.Get (printf "%s.png" $base)) (resources.Get (printf "favicons/%s.png" $name)) -}} {{- $found := false -}} {{- with $svg }} {{- $r := . | resources.Copy "/favicon.svg" }} {{- $found = true }} {{- end -}} {{- with $ico }} {{- $r := . | resources.Copy "/favicon.ico" }} {{- $found = true }} {{- end -}} {{- with $source }} {{- $img := . }} {{- range $sizes }} {{- $r := ($img.Resize (printf "%dx%d CatmullRom" . .)) | resources.Copy (printf "/favicon-%dx%d.png" . .) }} {{- $found = true }} {{- end }} {{- $apple := ($img.Resize "180x180 CatmullRom") | resources.Copy "/apple-touch-icon.png" }} {{- if not $noApplePrecomposed }} {{- /* Legacy iOS blind-probes this name even when a exists; publish a copy (no extra ). Disable via params.td.favicons.no_apple_precomposed. */}} {{- $_ := ($apple | resources.Copy "/apple-touch-icon-precomposed.png").RelPermalink }} {{- end }} {{- end -}} {{- if not $found }} {{- warnidf "docsy-no-favicons" "Docsy: no favicons found; add favicon.png (a square source image) and/or favicon.svg/.ico under assets/ (root or favicons/). See https://www.docsy.dev/docs/content/iconsimages/#add-your-favicons" -}} {{- end -}}
